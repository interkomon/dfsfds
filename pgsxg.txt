class MainActivity : AppCompatActivity() {
    lateinit var spinner:Spinner
    lateinit var storona1:EditText
    lateinit var storona2:EditText
    lateinit var radius:EditText
    lateinit var rasschet:Button
    lateinit var preferences: SharedPreferences

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
        spinner = findViewById(R.id.spinner)
        storona1 = findViewById(R.id.storona1)
        storona2 = findViewById(R.id.storona2)
        radius = findViewById(R.id.radius)
        rasschet = findViewById(R.id.radius)

        val vibor = ArrayAdapter.createFromResource(
            this,
            R.array.vibor,
            android.R.layout.simple_spinner_item
        ).apply {
            setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item)
        }
        spinner.adapter = vibor
        var perimetrSharedPreference = getPreferences(Context.MODE_PRIVATE)

    }

    fun rasschet(view: View) {

        val spin = spinner.selectedItem.toString()
        val st1 = storona1.text.toString().toDouble()
        val st2 = storona2.text.toString().toDouble()
        val rad = radius.text.toString().toDouble()

        var rach=0.0
        val perimetr = when(spin)
        {

            "Треугольник" ->

            rach=(st1+st2+rad)

            "Прямоугольник" ->

            rach= 2*(st1+st2)

            "Круг" ->
                rach = 2 * Math.PI * rad
            else -> 0.0
        }


    }
